# Core

Programa ::= SeqSentencias
SeqSentencias ::= Sentencia SeqSentencias
SeqSentencias ::= Sentencia
Sentencia ::= Asignacion
Asignacion ::= ID = ID
Asignacion ::= ID = Constante


# Definición de funciones

Sentencia ::= LlamadaFuncion
Sentencia ::= Funcion
Asignacion ::= ID = LlamadaFuncion
Asignacion ::= [ SeqID ] = LlamadaFuncion
Funcion ::= function ID ( SeqID ) SeqSentencias End
Funcion ::= function ID = ID ( SeqID ) SeqSentencias End
Funcion ::= function [ Parametros ] = ID ( SeqID ) SeqSentencias End
SeqID ::= ID , SeqID
SeqID ::= ID 
LlamadaFuncion ::= ID
LlamadaFuncion ::= ID ( )
LlamadaFuncion ::= ID ( Parametros )
Parametros ::= Parametros , Parametros
Parametros ::= ID
Parametros ::= Matriz
Parametros ::= Constante

# Definición de matrices

Asignacion ::= ID = Matriz
Matriz ::= [ Fila ]
Fila ::= SeqElementos ; Fila
Fila ::= SeqElementos 
SeqElementos ::= SeqElementos , SeqElementos 
SeqElementos ::= LlamadaFuncion
SeqElementos ::= Constante
SeqElementos ::= ID


# Estructuras de control

Sentencia ::= Condicional
Sentencia ::= DefFor
Sentencia ::= DefWhile
DefFor ::= for ID = Termino : Termino SeqSentencias End
DefWhile ::= While Termino SeqSentencias End
Condicional ::= if ( Termino ) End
Condicional ::= if ( Termino ) SeqSentencias End
Condicional ::= if ( Termino ) SeqSentencias else SeqSentencias End
Condicional ::= if ( Termino ) SeqSentencias CondicionalAnidado End
Condicional ::= if ( Termino ) SeqSentencias CondicionalAnidado else SeqSentencias End
CondicionalAnidado ::= CondicionalAnidado elseif ( Termino ) SeqSentencias
CondicionalAnidado ::= elseif ( Termino ) SeqSentencias

# Expresiones numericas

Asignacion ::= ID = Termino
Termino ::= True
Termino ::= False
Termino ::= Constante
Termino ::= Matriz
Termino ::= ID
Termino ::= LlamadaFuncion 
Termino ::= ( Termino )  


# Expresiones lógicas

Termino ::= ExpLogica
ExpLogica ::= ( ExpLogica )
ExpLogica ::= True
ExpLogica ::= False
ExpLogica ::= Termino OpLog Termino
ExpLogica ::= Termino OpLog ExpLogica
ExpLogica ::= ExpLogica OpLog Termino
OpLog ::= &&
OpLog ::= ||
OpLog ::= <
OpLog ::= >
OpLog ::= >=
OpLog ::= <=
OpLog ::= ==
OpLog ::= ~=


#Expresiones Aritmeticas

Termino ::= ExpAritmetica
ExpAritmetica ::= ( ExpAritmetica ) 
ExpAritmetica ::= Termino OpAritmetico Termino
ExpAritmetica ::= Termino OpAritmetico ExpAritmetica
ExpAritmetica ::= ExpAritmetica OpAritmetico Termino
OpAritmetico ::= +
OpAritmetico ::= -
OpAritmetico ::= *
OpAritmetico ::= .*
OpAritmetico ::= ./
OpAritmetico ::= .\
OpAritmetico ::= /
OpAritmetico ::= \
OpAritmetico ::= .^
OpAritmetico ::= ^

# Core

Programa ::= SeqSentencias
SeqSentencias ::= Sentencia SeqSentencias
SeqSentencias ::= Sentencia
Sentencia ::= Asignacion


# FuncDef Definición de funciones

Sentencia ::= LlamadaFuncion
Sentencia ::= Funcion
Asignacion ::= ID = LlamadaFuncion
Asignacion ::= [ SeqID ] = LlamadaFuncion
IniFuncion ::= function ID ( SeqID )
IniFuncion ::= function ID = ID ( SeqID )
IniFuncion ::= function [ Parametros ] = ID ( SeqID )
Funcion ::= IniFuncion SeqSentencias End
SeqID ::= ID , SeqID
SeqID ::= ID 
LlamadaFuncion ::= ID
LlamadaFuncion ::= ID ( )
LlamadaFuncion ::= ID ( Parametros )
Parametros ::= Parametros , Parametros
Parametros ::= Termino

# MatDef Definición de matrices

Matriz ::= [ Fila ]
Fila ::= SeqElementos ; Fila
Fila ::= SeqElementos 
SeqElementos ::= SeqElementos , SeqElementos 
SeqElementos ::= LlamadaFuncion
SeqElementos ::= Constante
SeqElementos ::= ID


# CtrlStr Estructuras de control

Sentencia ::= Condicional
Sentencia ::= DefFor
Sentencia ::= DefWhile
IniFor ::= for ID = Termino : Termino
DefFor ::=  IniFor SeqSentencias End
IniWhile ::= While Termino
DefWhile ::= IniWhile SeqSentencias End
IniCondicional ::= if ( Termino )
Condicional ::= IniCondicional End
Condicional ::= IniCondicional SeqSentencias End
Condicional ::= IniCondicional SeqSentencias else SeqSentencias End
Condicional ::= IniCondicional SeqSentencias CondicionalAnidado End
Condicional ::= IniCondicional SeqSentencias CondicionalAnidado else SeqSentencias End
CondicionalAnidado ::= CondicionalAnidado elseif ( Termino ) SeqSentencias
CondicionalAnidado ::= elseif ( Termino ) SeqSentencias

# NumExp Expresiones numericas

Asignacion ::= ID = Termino
Termino ::= Constante
Termino ::= Matriz
Termino ::= ID
Termino ::= LlamadaFuncion 
Termino ::= ( Termino )  


# LogExp Expresiones lógicas

Termino ::= ExpLogica
ExpLogica ::= ( ExpLogica )
ExpLogica ::= Constante
ExpLogica ::= Termino OpLog Termino
ExpLogica ::= Termino OpLog ExpLogica
ExpLogica ::= ExpLogica OpLog Termino
OpLog ::= &&
OpLog ::= ||
OpLog ::= <
OpLog ::= >
OpLog ::= >=
OpLog ::= <=
OpLog ::= ==
OpLog ::= ~=


# AritExp Expresiones Aritmeticas

Termino ::= ExpAritmetica
ExpAritmetica ::= ( ExpAritmetica ) 
ExpAritmetica ::= Termino OpAritmetico Termino
ExpAritmetica ::= Termino OpAritmetico ExpAritmetica
ExpAritmetica ::= ExpAritmetica OpAritmetico Termino
OpAritmetico ::= +
OpAritmetico ::= -
OpAritmetico ::= *
OpAritmetico ::= .*
OpAritmetico ::= ./
OpAritmetico ::= .\
OpAritmetico ::= /
OpAritmetico ::= \
OpAritmetico ::= .^
OpAritmetico ::= ^
